namespace = puppet_management

puppet_management.1 = {
	type = country_event
	hidden = yes

	trigger = {
		any_country = {
			is_subject = yes
			has_law_or_variant = law_type:law_council_republic
			overlord ?= {
				NOT = {
					has_law_or_variant = law_type:law_council_republic
				}
			}
		}
	}

	immediate = {
		random_country = {
			limit = {
				is_subject = yes
				has_law_or_variant = law_type:law_council_republic
				overlord ?= {
					NOT = {
						has_law_or_variant = law_type:law_council_republic
					}
				}
			}
			change_relations = {
				country = overlord
				value = -200
			}
			make_independent = yes
		}
	}
}
